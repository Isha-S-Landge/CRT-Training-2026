5. we cannot multiply or divide pointer value but we can add and subtract pointer.

#include <iostream>
int main() {
    char b='x',*pc=&b;
    int a=12,*pi=&a;
    float c=2.5,*pf=&c;
    printf("value of pi=address of a=%u\n",pi);
    printf("value of pc=address of a=%u\n",pc);
    printf("value of pf=address of a=%u\n",pf);
    pi++,pc++,pf++;
    printf(" Now value of pi==%u\n",pi);
    printf(" Now value of pc==%u\n",pc);
    printf(" Now value of pf==%u\n",pf);
    
    return 0;
}

6. print value of array using pointer and subscript notation 

#include <iostream>
int main() {
    int arr[5]={5,10,20,25,30};
    for(int i=0;i<5;i++){
        printf("value of a[%d]\n",i);
        printf("%d\n",arr[i]);
        printf("%d\n",i[arr]);
        printf("%d\n",*(arr+i));
        printf("%d\n",*(i+arr));

    return 0;
