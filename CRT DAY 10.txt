Binary Tree


#include <iostream>
using namespace std;
class node{
    public:
    int data;
    node*right;
    node*left;
    node(int value){
        data=value;
        left-NULL;
        right=NULL;
    }
};

int main() {
    node*root= new node(10);
    root->left=new node(20);
    root->right=new node(30);
    
    root->left->left=new node(40);
    root->left->right=new node(40);
    cout<<"root node:"<<root->data<<endl;
    return 0;

}
