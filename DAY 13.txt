#include <iostream>
#include <vector>

using namespace std;

int main() {
    vector<int> a = {1, 2, 3, 4, 5};
    vector<int> b = {3, 4, 5, 6, 7};

    int i = 0, j = 0;

    while (i < a.size() && j < b.size()) {
        if (a[i] == b[j]) {
            cout << a[i] << " ";
            i++;
            j++;
        } else if (a[i] < b[j]) {
            i++;
        } else {
            j++;
        }
    }

    return 0;
}

#include <iostream>
#include <vector>

using namespace std;

class Graph{
    private:
    bool**adjMatrix;// pointer to
    int vertices;
    public:
    //constructure
    Graph(int vertices){
        this->vertices=vertices;
        
        adjMatrix=new bool*[vertices];
        for(int i=0;i<vertices;i++){
            adjMatrix[i]=new bool[vertices];
            for (int j=0;j<vertices;j++){
                adjMatrix[i][j]=false;
            }
        }
    }
    void addEdge(int source ,int destination){
        adjMatrix[source][destination]=true;
        adjMatrix[destination][source]=true;
    }
    
    void display(){
        cout<<"\nAdjancey matrix\n";
        for(int i=0;i<vertices;i++){
            for(int j=0;j<vertices;j++){
                cout<<adjMatrix[i][j];         
                }
        cout<<endl;
        }
    }
    
};

int main() {
    Graph g(6);
    g.addEdge(0,1);
    g.addEdge(0,2);
    g.addEdge(1,3);
    g.addEdge(2,3);
    g.addEdge(3,4);
    g.addEdge(4,5);
    g.display();

    return 0;
}
